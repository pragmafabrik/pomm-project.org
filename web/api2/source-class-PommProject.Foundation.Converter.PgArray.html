 35:     {
 36:         if ($data === null || $data === 'NULL') {
 37:             return null;
 38:         }
 39: 
 40:         if ($data !== "{}") {
 41:             $converter = $this->getSubtypeConverter($type, $session);
 42:

 44:                     return $val !== "NULL" ? $converter->fromPg($val, $type) : null;
 45:                 }, str_getcsv(trim($data, "{}")));
 46:         } else {
 47:             return [];
 48:         }
 49:     }

 52:      * @see ConverterInterface
 53:      */
 54:     public function toPg($data, $type, Session $session)
 55:     {
 56:         if ($data === null) {
 57:                 return sprintf("NULL::%s[]", $type);
 58:         }
 59: 
 60:         $converter = $this->getSubtypeConverter($type, $session);
 61:         $data = $this->checkArray($data);
 62: 
 63:         return sprintf('ARRAY[%s]::%s[]', join(',', array_map(function ($val) use ($converter, $type) {
 64:                     return $converter->toPg($val, $type);
 65:                 }, $data)), $type);
 66:     }

 69:      * checkArray

 73:      * @access protected
 74:      * @param  mixed    $data
 75:      * @return array    $data
 76:      */
 77:     protected function checkArray($data)
 78:     {
 79:         if (!is_array($data)) {
 80:             throw new ConverterException(
 81:                 sprintf(
 82:                     "Array converter data must be an array ('%s' given).",
 83:                     gettype($data)

 85:             );
 86:         }
