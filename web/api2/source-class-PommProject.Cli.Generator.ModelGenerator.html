 36:      * generate
 37:      *
 38:      * Generate structure file.

 40:      * @access public
 41:      * @param  InputInterface  $input
 42:      * @param  OutputInterface $output
 43:      * @return null
 44:      */
 45:     public function generate(InputInterface $input, OutputInterface $output)
 46:     {
 47:         $schema_oid = $this
 48:             ->getSession()
 49:             ->getInspector()
 50:             ->getSchemaOid($this->schema);
 51: 
 52:         if ($schema_oid === null) {
 53:             throw new GeneratorException(sprintf("Schema '%s' does not exist.", $this->schema));
 54:         }
 55: 
 56:         $relations_info = $this
 57:             ->getSession()
 58:             ->getInspector()
 59:             ->getSchemaRelations($schema_oid, new Where('cl.relname = $*', [$this->relation]))
 60:             ;
 61: 
 62:         if ($relations_info->isEmpty()) {
 63:             throw new GeneratorException(sprintf("Relation '%s.%s' does not exist.", $this->schema, $this->relation));
 64:         }
 65: 
 66:         if (file_exists($this->filename) && !$input->getOption('force')) {
 67:             throw new GeneratorException(sprintf("Cannot overwrite file '%s' without --force option.", $this->filename));
 68:         }
 69: 
 70:         $this->outputFileCreation($output);
 71: 
 72:         $this->saveFile(
 73:             $this->filename,
 74:             $this->mergeTemplate(
 75:                 [
 76:                     'entity'        => Inflector::studlyCaps($this->relation),
 77:                     'namespace'     => trim($this->namespace, '\\'),
 78:                     'trait'         => $relations_info->current()['type'] === 'table' ? 'WriteQueries' : 'ReadQueries',
 79:                     'relation_type' => $relations_info->current()['type'],
 80:                     'relation'      => $this->relation
 81:                 ]
 82:             )
 83:         );
